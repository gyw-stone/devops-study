- name: Modify DNS configuration on target hosts
  hosts: all  # 选择你要运行的目标主机
  become: yes  # 如果需要 root 权限，使用 become

  tasks:
    - name: Copy the DNS configuration file as backup if not exists
      ansible.builtin.stat:
        path: /etc/resolv.conf.bak
      register: resolv_bak_stat

    - name: Backup the resolv.conf file if backup doesn't exist
      ansible.builtin.copy:
        src: /etc/resolv.conf
        dest: /etc/resolv.conf.bak
        remote_src: yes
      when: resolv_bak_stat.stat.exists == False

    - name: Remove all existing nameserver entries
      ansible.builtin.replace:
        path: /etc/resolv.conf
        regexp: '^nameserver.*'
        replace: ''
        backup: yes
    
    - name: Ensure the DNS server is set to 172.26.22.110
      ansible.builtin.lineinfile:
        path: /etc/resolv.conf
        line: 'nameserver 172.26.22.110'
        state: present
        create: yes

    - name: Ensure a "mideng" file exists to track state
      ansible.builtin.file:
        path: /etc/mideng
        state: touch
      register: mideng_stat

    - name: Check if "mideng" file exists
      debug:
        msg: "Mideng file already exists"
      when: mideng_stat.changed == False

    - name: Output DNS modification status
      debug:
        msg: "DNS server has been updated successfully."
